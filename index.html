    JavaScript Raytracer

    • Demo (use arrow keys to move around)
    • Include the script in your webpage 
    <script src="https://rawgit.com/MichaelSnowden/raytracer.js/master/raytracer.js"></script>
    • Create a raytracer with some properties
    raytracer = new Raytracer({parent: document.getElementById("raytracerContainer")});
    • Render the scene
    raytracer.render()
    • Add some controls
    document.onkey = checkKey;

function checkKey(e) {
    e = e || window.event;
    e.preventDefault();

    if (e.keyCode == '38') {
        raytracer.camera[2] += 0.1;
    }
    else if (e.keyCode == '40') {
        raytracer.camera[2] -= 0.1;
    }
    else if (e.keyCode == '37') {
       raytracer.camera[0] -= 0.1;
    }
    else if (e.keyCode == '39') {
       raytracer.camera[0] += 0.1;
    }
    raytracer.render();
}
